  public static void setJframe() throws RuntimeException {
    Dimension dim = Toolkit.getDefaultToolkit().getScreenSize();
    this.setLocation(
        dim.width / 2 - this.getSize().width / 2,
        dim.height / 2 - this.getSize().height / 2
      );
  }
}
